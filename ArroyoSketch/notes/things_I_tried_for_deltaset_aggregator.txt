-- Simplified UDFs for tracking set differences of concat_ws strings
/*
[dependencies]
serde = { version = "1.0", features = ["derive"] }
serde_json = "1.0"
*/

use arroyo_udf_plugin::udf;
use std::collections::HashSet;

-- UDAF to collect unique concat_ws strings in a window
#[udf]
fn collect_unique_strings(strings: Vec<String>) -> String {
    let unique_strings: HashSet<String> = strings.into_iter()
        .filter(|s| !s.is_empty())
        .collect();

    let mut sorted_strings: Vec<String> = unique_strings.into_iter().collect();
    sorted_strings.sort();

    if sorted_strings.is_empty() {
        String::new()
    } else {
        sorted_strings.join("|")
    }
}

-- UDF to find set differences between two pipe-separated string sets
#[udf]
fn find_set_differences(current_set: &str, previous_set: &str) -> String {
    let current: HashSet<&str> = if current_set.is_empty() {
        HashSet::new()
    } else {
        current_set.split('|').collect()
    };

    let previous: HashSet<&str> = if previous_set.is_empty() {
        HashSet::new()
    } else {
        previous_set.split('|').collect()
    };

    let mut changes = Vec::new();

    // Find new strings (in current but not in previous)
    for item in &current {
        if !previous.contains(item) {
            changes.push(format!("NEW:{}", item));
        }
    }

    // Find missing strings (in previous but not in current)
    for item in &previous {
        if !current.contains(item) {
            changes.push(format!("MISSING:{}", item));
        }
    }

    if changes.is_empty() {
        "NO_CHANGES".to_string()
    } else {
        changes.join("|")
    }
}

-- UDF to check if there are any changes
#[udf]
fn has_changes(differences: &str) -> bool {
    differences != "NO_CHANGES" && !differences.is_empty()
}

CREATE TABLE string_set_changes (
    window_start TIMESTAMP,
    window_end TIMESTAMP,
    changes TEXT,
    current_set TEXT,
    previous_set TEXT
) WITH (
    connector = 'kafka',
    bootstrap_servers = 'localhost:9092',
    topic = 'set-changes',
    format = 'json'
);

-- Main query using LAG to compare consecutive windows
WITH windowed_sets AS (
    SELECT
        window_start(TUMBLE(interval '10 seconds')) as window_start,
        window_end(TUMBLE(interval '10 seconds')) as window_end,
        collect_unique_strings(
            array_agg(concat_ws(';', label_env, label_service, label_instance, label_job))
        ) as current_set
    FROM metrics_data
    GROUP BY TUMBLE(interval '10 seconds')
),
with_previous AS (
    SELECT
        window_start,
        window_end,
        current_set,
        LAG(current_set, 1) OVER (ORDER BY window_start) as previous_set
    FROM windowed_sets
),
changes_detected AS (
    SELECT
        window_start,
        window_end,
        current_set,
        COALESCE(previous_set, '') as previous_set,
        find_set_differences(current_set, COALESCE(previous_set, '')) as changes
    FROM with_previous
)
INSERT INTO string_set_changes
SELECT
    window_start,
    window_end,
    changes,
    current_set,
    previous_set
FROM changes_detected
WHERE has_changes(changes);

WITH current_window AS (
    SELECT
        window_start(TUMBLE(interval '10 seconds')) as window_start,
        window_end(TUMBLE(interval '10 seconds')) as window_end,
        collect_unique_strings(
            array_agg(concat_ws(';', label_env, label_service, label_instance, label_job))
        ) as current_set
    FROM metrics_data
    GROUP BY TUMBLE(interval '10 seconds')
),
with_state AS (
    SELECT
        cw.window_start,
        cw.window_end,
        cw.current_set,
        COALESCE(ss.last_set, '') as previous_set
    FROM current_window cw
    LEFT JOIN set_state ss ON ss.state_key = 'current'
)
SELECT
    window_start,
    window_end,
    find_set_differences(current_set, previous_set) as changes,
    current_set,
    previous_set
FROM with_state
WHERE has_changes(find_set_differences(current_set, previous_set));
